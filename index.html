        // CALCOLO PREZZI dinamico

        function calculatePrice() {

            const checkin = new Date(document.getElementById('checkin').value);

            const checkout = new Date(document.getElementById('checkout').value);

            const guests = parseInt(document.getElementById('guests').value);

            const linenPerPerson = parseInt(document.getElementById('linen').value);

            

            // Validazione date

            if (!checkin.getTime() || !checkout.getTime() || checkout <= checkin) {

                return;

            }

            

            // Calcolo notti

            const diffTime = Math.abs(checkout - checkin);

            const nights = Math.ceil(diffTime / (1000 * 60 * 60 * 24));

            

            // Calcolo stagione

            let seasonName = '';

            let pricePerNight = 0;

            let seasonColor = '';

            

            const month = checkin.getMonth(); // 0-11

            const day = checkin.getDate();

            

            // Logica stagionale semplificata

            const isHighSeason = (month === 6) || (month === 7) || // Luglio(6) e Agosto(7)

                                (month === 11 && day >= 20) || // Natale

                                (month === 0 && day <= 6); // Capodanno

            

            const isMidSeason = (month === 5) || // Giugno

                               (month === 8) || // Settembre

                               (month === 3 && (day >= 20 && day <= 25)) || // Pasqua/aprile

                               (month === 4) || // Maggio (ponte 1 mag)

                               (month === 10 && day === 1); // tutti i santi

            

            if (isHighSeason) {

                seasonName = 'Alta Stagione';

                pricePerNight = 380;

                seasonColor = '#e74c3c';

            } else if (isMidSeason) {

                seasonName = 'Media Stagione';

                pricePerNight = 220;

                seasonColor = '#f39c12';

            } else {

                seasonName = 'Bassa Stagione';

                pricePerNight = 140;

                seasonColor = '#2ecc71';

            }

            

            // Calcolo totale

            const accommodationTotal = nights * pricePerNight;

            const cleaningFee = 100; // Pulizia obbligatoria

            const linenTotal = guests * linenPerPerson;

            const total = accommodationTotal + cleaningFee + linenTotal;

            

            // Visualizzazione risultato

            const resultBox = document.getElementById('resultBox');

            const totalPrice = document.getElementById('totalPrice');

            const priceDetails = document.getElementById('priceDetails');

            

            resultBox.classList.add('show');

            totalPrice.textContent = '€' + total.toLocaleString('it-IT');

            

            let detailsHTML = `

                <strong>${nights} notti</strong> in ${seasonName} (${pricePerNight}€/notte)<br>

                Alloggio: €${accommodationTotal} | Pulizia: €${cleaningFee}

                ${linenTotal > 0 ? `<br>Biancheria (${guests} persone): €${linenTotal}` : ''}

            `;

            

            // Soggiorno minimo check

            let minNights = 3;

            if (isHighSeason) minNights = 7;

            else if (isMidSeason) minNights = 5;

            

            if (nights < minNights) {

                detailsHTML += `<br><span style="color: #ffeb3b; font-weight: bold;"><i class="fas fa-exclamation-triangle"></i> Minimo ${minNights} notti richiesto per questa stagione</span>`;

                totalPrice.style.opacity = '0.6';

            } else {

                totalPrice.style.opacity = '1';

            }

            

            priceDetails.innerHTML = detailsHTML;

        }


        // Imposta date minime (oggi)

        const today = new Date().toISOString().split('T')[0];

        document.getElementById('checkin').min = today;

        document.getElementById('checkout').min = today;

    </script>
